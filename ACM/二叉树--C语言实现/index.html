#ifndef _BINARYTREE_H
#define _BINARYTREE_H
typedef char TElemType;
const int MAXLENGTH = 100;
typedef struct node *treePtr;   //the pointer of tree node
struct node{    //the struct of tree node
    TElemType data;
    bool ltag,rtag;
    treePtr lchild,rchild;
};

/*
    Recurrence create one binaryTree
    '#' is NULL
*/ 
treePtr CreateBT1();
/*
    NoRecurrence create one binaryTree
    input the key and data
*/
treePtr CreateBT2(); 
/*
    print the tree use generalized list
*/
void printBT(treePtr T);
/*
    print the threadingTree use generalized list
*/
void printThBT(treePtr T);

/*
    Traverse the binaryTree use recursion or not
*/
void PreOrder(treePtr T);
void InOrder(treePtr T);
void PostOrder(treePtr T);
void PreOrderNoRe(treePtr T);
void InOrderNoRe(treePtr T);
void PostOrderNoRe(treePtr T);
void LeverOrder(treePtr T);

/*
    Threading the binaryTree of different order     
*/
void InThreading(treePtr T);
void PreThreading(treePtr T);
void PostThreading(treePtr T);

/*
    Add the head node
    Threading T to Thrt in the way of Threading
    Theading is a function pointer 
*/
treePtr OrderThreading(treePtr T,void(*Threading)(treePtr T));

/*  
    Find the next node in ThreadingTree
*/
treePtr InNext(treePtr T);
treePtr PreNext(treePtr T);
treePtr PostNext(treePtr T);

/*
    Traverse the ThreadingTree
    next is a function pointer  
*/
void ThreadingTraverse(treePtr T,treePtr (*next)(treePtr T));

/*
    cancel the thread of binaryTree
    use in printThBT()
*/
void CancelThreading(treePtr T);

#endif

void printBT(treePtr T){
    if(T){
        cout<<T->data;
        if(T->lchild || T -> rchild){
            cout<<"(";
            printBT(T->lchild);
            if(T->rchild)
                cout<<",";
            printBT(T->rchild);
            cout<<")";
        }
    }
}   

void printThBT(treePtr T){
    if(T){
        cout<<T->data;
        cout<<"(";
        if(T->ltag){

            cout<<",Link:";
            printThBT(T->rchild);

            cout<<",Thread:"<<T->rchild->data;

        cout<<")";
    }
}

void PreOrderNoRe(treePtr T){
    treePtr s[MAXLENGTH];
    int top = -1;
    while(T != NULL || top != -1){
        while(T != NULL){
            cout<<T->data<<" ";

            T = T -> lchild;
        }
        if (top != -1){
            T = s[top--];
            T = T -> rchild;
        }
    }

void InOrderNoRe(treePtr T){

            T = T -> lchild;

void PostOrderNoRe(treePtr T){
    int top = -1;
    stk s[MAXLENGTH];
    while(T != NULL || top != -1){
        while(T != NULL){
            s[++top].ptr = T;
            s[top].flag = 1;
            T = T -> lchild;            
        }
        while(top != -1 && s[top].flag == 2){
            T = s[top--].ptr;
            cout<<T->data<<" ";
        }
        if(top != -1){
            s[top].flag = 2;
            T = s[top].ptr->rchild;
        }else break;
    }
}

void LeverOrder(treePtr T){
    int front, rear;
    front = rear = 0;
    treePtr q[MAXLENGTH];
    if(T == NULL) return ;
    q[++rear] = T;
    while(front != rear){
        T = q[++front];
        cout <<T->data<<" ";
        if(T->lchild != NULL)
            q[++rear] = T->lchild;
        if(T->rchild != NULL)
            q[++rear] = T->rchild;
    }
}

treePtr pre;


treePtr OrderThreading(treePtr T,void(*Threading)(treePtr T)){
    treePtr Thrt;   //head pointer
    if(!(Thrt = (treePtr)new(node))) 
        return NULL;
    Thrt -> ltag = false;    //init the head node
    Thrt -> rtag = true;
    Thrt -> rchild = Thrt;
    if(!T) 
        Thrt -> lchild = Thrt;  
    else{

        Thrt -> lchild = T;
        Thrt -> ltag = true;
        pre = Thrt;
        Threading(T);   //Threading binaryTree
        if(Threading != PostThreading){   //PostThreading the last node is root
            pre -> rchild = Thrt;
            pre -> rtag = false;
        }else{
            if(!T->rchild){  //while posthreading and the root doesn't have rchild
                T->rchild = Thrt;
                T->rtag = false;
            }
        }
    }
    Head = Thrt;
    return Thrt;
}

void ThreadingTraverse(treePtr T,treePtr(*next)(treePtr T)){
    treePtr tmp = T;
    for(;;){
        tmp = next(tmp);
        if(tmp == T) break;
        cout<<tmp->data<<" ";
    }
}


void PreThreading(treePtr T){
    if(T){
        if(!T->lchild){     //predecessor of T
            T->ltag = false;
            T->lchild = pre;
        }
        if(!pre->rchild){   //successor of pre
            pre->rtag = false;
            pre->rchild = T;
        }
        pre = T;
        if(T->ltag)  
            PreThreading(T->lchild);
        if(T->rtag)
            PreThreading(T->rchild);
    }
}

void InThreading(treePtr T){
   if(T){
        InThreading(T->lchild);
        if(!T->lchild){     
            T->lchild = pre;
            T->ltag = false;
        }
        if(!pre->rchild){   
            pre->rchild = T;
            pre->rtag = false;
        }
        pre = T;
        InThreading(T->rchild);
    }
}

void PostThreading(treePtr T){
    if(T){
        PostThreading(T->lchild);
        PostThreading(T->rchild);
        if(!T->lchild){
            T->lchild = pre;
            T->ltag = false;
        }
        if(!pre->rchild){
            pre->rchild = T;
            pre->rtag = false;
        }
        pre = T;
    }
}


treePtr PreNext(treePtr T){
    if(T->ltag)
        return T->lchild;
    return T->rchild;
}

treePtr InNext(treePtr T){
    treePtr tmp;
    tmp = T-> rchild;
    if(T->rtag)
        while(tmp->ltag)
            tmp = tmp->lchild;
    return tmp;
}

/*
    find the parent node of node from binaryTree T
*/
treePtr parent(treePtr T,treePtr p){
    treePtr tmp = T;
    if(tmp -> lchild == p)
        return tmp;
    else{
        tmp = tmp -> lchild;
        while(tmp -> lchild != p && tmp -> rchild != p){
            if(tmp->rtag)
                tmp = tmp -> rchild;
            else
                tmp = tmp -> lchild;
        }
    }
    return tmp;
}

treePtr PostNext(treePtr T){
    treePtr tmp = T;
    if(tmp == Head){
        tmp = tmp -> lchild;
        if(tmp == Head)   //T is empty
            return tmp;
        for(;;){
            while(tmp -> ltag)
                tmp = tmp->lchild;
            if(tmp->rtag)
                tmp = tmp->rchild;
            else
                return tmp;
        }
    }
    treePtr par = parent(Head,tmp);
    if(par == Head)
        tmp = Head;
    else if(!par->rtag || tmp == par->rchild)
        tmp = par; 
    else{
        while(par -> rtag){
            par = par -> rchild;
            while(par-> ltag)
                par = par -> lchild;
        }
        tmp = par;
    }
    return tmp;
}



void CancelThreading(treePtr T){
    int front,rear;
    front = rear = 0;
    treePtr q[MAXLENGTH];
    if(T == NULL) return;
    q[++rear] = T;
    while(front != rear){
        T = q[++front];
        if(T->ltag)
            q[++rear] = T->lchild;
        else
            T->lchild = NULL;
        if(T->rtag)
            q[++rear] = T->rchild;
        else
            T->rchild = NULL;
    }
}
